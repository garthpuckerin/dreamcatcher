**2. Implement Tag Suggestion Endpoint**

Create `api/ai/suggest-tags.js`:

```javascript
import OpenAI from 'openai';

// Rate limiting helper
const rateLimit = new Map();

function checkRateLimit(identifier) {
  const now = Date.now();
  const userRequests = rateLimit.get(identifier) || [];
  const recentRequests = userRequests.filter(time => now - time < 60000); // 1 minute

  if (recentRequests.length >= 10) {
    return false;
  }

  recentRequests.push(now);
  rateLimit.set(identifier, recentRequests);
  return true;
}

export default async function handler(req, res) {
  // CORS headers
  res.setHeader('Access-Control-Allow-Origin', '*');
  res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS');
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type');

  if (req.method === 'OPTIONS') {
    return res.status(200).end();
  }

  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  // Rate limiting
  const identifier = req.headers['x-forwarded-for'] || 'unknown';
  if (!checkRateLimit(identifier)) {
    return res.status(429).json({ error: 'Rate limit exceeded' });
  }

  try {
    const { dreamContent } = req.body;

    if (!dreamContent || dreamContent.length < 10) {
      return res.status(400).json({ error: 'Invalid dream content' });
    }

    // Initialize OpenAI
    const openai = new OpenAI({
      apiKey: process.env.OPENAI_API_KEY
    });

    // Call OpenAI
    const response = await openai.chat.completions.create({
      model: 'gpt-3.5-turbo',
      messages: [
        {
          role: 'system',
          content: 'You are a helpful assistant that suggests relevant tags for project ideas. Return 5-8 concise tags as a JSON array.'
        },
        {
          role: 'user',
          content: `Suggest tags for this project idea:\n\n${dreamContent}`
        }
      ],
      temperature: 0.7,
      max_tokens: 100
    });

    const content = response.choices[0].message.content;
    const tags = JSON.parse(content);

    return res.status(200).json({ tags });
  } catch (error) {
    console.error('OpenAI API error:', error);
    return res.status(500).json({ error: 'AI service temporarily unavailable' });
  }
}
```

  content: z.string().max(50000),
  source: z.string().max(100).optional(),

// Sanitize HTML content
export function sanitizeHTML(dirty) {
  return DOMPurify.sanitize(dirty, {

### 2.2 Rate Limiting (Day 4)

**Goal:** Prevent abuse and API overuse

#### Implementation

**1. Backend Rate Limiting (Already in API proxy)**

See Section 1.3 - checkRateLimit function

**2. Frontend Rate Limiting**

Create `src/lib/rateLimiter.js`:

```javascript
class RateLimiter {
  constructor(maxRequests, windowMs) {
    this.maxRequests = maxRequests;
    this.windowMs = windowMs;
    this.requests = new Map();
  }

  tryAcquire(key) {
    const now = Date.now();
    const userRequests = this.requests.get(key) || [];
    const recentRequests = userRequests.filter(time => now - time < this.windowMs);

    if (recentRequests.length >= this.maxRequests) {
      return false;
    }

    recentRequests.push(now);
    this.requests.set(key, recentRequests);
    return true;
  }

  getRemainingTime(key) {
    const userRequests = this.requests.get(key) || [];
    if (userRequests.length === 0) return 0;

    const oldestRequest = Math.min(...userRequests);
    const elapsed = Date.now() - oldestRequest;
    return Math.max(0, this.windowMs - elapsed);
  }
}

// Export rate limiters for different operations
export const aiRequestLimiter = new RateLimiter(5, 60000); // 5 per minute
export const searchLimiter = new RateLimiter(20, 60000); // 20 per minute
```

**3. Apply to AI Features**

Update `src/hooks/useAI.js`:

import { aiRequestLimiter } from '../lib/rateLimiter';

export function useAI() {
  const suggestTags = async (content) => {

## Data Flow: Creating a Dream

1. User captures conversation
2. Frontend validates input
3. API proxy receives request
4. Rate limiter checks quota
5. Data sanitized and validated
6. Saved to Supabase/LocalStorage
7. AI analysis triggered (async)
8. Tags and summary generated
9. UI updated with new dream

### 5.3 Contributing Guide (Day 13)

**Goal:** Make it easy for others to contribute

Create `CONTRIBUTING.md`:

```markdown
# Contributing to Dreamcatcher

Thank you for considering contributing to Dreamcatcher!

## Development Setup

1. Fork the repository
2. Clone your fork
3. Install dependencies: `npm install`
4. Create a feature branch: `git checkout -b feature/amazing-feature`
5. Make your changes
6. Run tests: `npm run test`
7. Commit: `git commit -m 'feat: add amazing feature'`
8. Push: `git push origin feature/amazing-feature`
9. Open a Pull Request

## Commit Convention

We follow [Conventional Commits](https://www.conventionalcommits.org/):

- `feat:` New feature
- `fix:` Bug fix
- `docs:` Documentation
- `refactor:` Code refactoring
- `test:` Adding tests
- `chore:` Maintenance

## Code Style

- Run `npm run lint` before committing
- Run `npm run format` to auto-format
- Follow existing code patterns

## Pull Request Process

1. Update README if needed
2. Add tests for new features
3. Ensure all tests pass
4. Update CHANGELOG.md
5. Request review from maintainers

## Questions?

Open an issue or reach out to [contact info]
```

## Risk Mitigation

### Potential Issues & Solutions

| Risk | Impact | Mitigation |
|------|--------|------------|
| Vercel deployment fails | High | Test locally with `vercel dev` first |
| API proxy doesn't work | High | Test endpoints with curl/Postman |
| Tests don't reach 60% | Medium | Focus on critical paths only |
| Screenshots look unprofessional | Medium | Use Figma/Canva for polish |
| Video recording quality poor | Low | Use Loom or OBS Studio |
| Mobile responsive issues | Medium | Test early on multiple devices |
| Performance below target | Medium | Implement lazy loading, code splitting |

### Contingency Plans

**If behind schedule:**
1. Skip Phase 3.3 (UI polish) - can be done later
2. Reduce test coverage target to 50%
3. Skip architecture diagrams
4. Use simpler demo video

**If deployment issues:**
